import pygame as pg
from pygame import *
from bird import Bird
from pipes import Pipes

def main():
  def update(screen):
    bird.update(screen, events)
    pipes.update(screen)
    
  pg.init()

  pg.display.set_caption('FlappyBird')

  fps = 60

  screen_width = 800
  screen_height = 600
 
  screen = pg.display.set_mode((screen_width, screen_height))

  clock = pg.time.Clock()

  bird = Bird(screen_width, screen_height)
  pipes = Pipes(screen_width, screen_height)

  running = True
  while running:
    events = pg.event.get()
    for event in pg.event.get():
      if event.type == QUIT:
        running = False

    screen.fill((0, 0, 0))
    
    update(screen)

    pg.display.flip()
    
    clock.tick(fps)

  pg.quit()

if __name__ == "__main__":
  main()
